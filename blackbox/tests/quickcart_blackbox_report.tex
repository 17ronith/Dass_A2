\section*{Overview}
This report documents black-box testing for the QuickCart REST API based on the provided documentation. The tests validate correct HTTP status codes, JSON response structures, and data correctness according to the specification. Tests were automated using \texttt{pytest} and \texttt{requests}.

\section*{Environment}
\begin{itemize}[leftmargin=*]
  \item Base URL: \texttt{http://localhost:8080/api/v1}
  \item Headers used: \texttt{X-Roll-Number} and \texttt{X-User-ID} (user-scoped endpoints)
  \item Tools: \texttt{pytest}, \texttt{requests}
\end{itemize}

\section*{Bug Reports}
Each bug includes endpoint, request, expected result, and actual result observed.

\subsection*{Bug 1: Address creation rejects valid payload}
\textbf{Endpoint:} POST \texttt{/api/v1/addresses} \\
\textbf{Request:} Headers: \texttt{X-Roll-Number: 123}, \texttt{X-User-ID: 799}; Body: \{label: HOME, street: 123 Main Street, city: Metropolis, pincode: 123456, is\_default: true\} \\
\textbf{Expected:} 200 with created address object \\
\textbf{Actual:} 400 Bad Request

\subsection*{Bug 2: Cart add accepts quantity 0}
\textbf{Endpoint:} POST \texttt{/api/v1/cart/add} \\
\textbf{Request:} Headers: \texttt{X-Roll-Number: 123}, \texttt{X-User-ID: 799}; Body: \{product\_id: 1, quantity: 0\} \\
\textbf{Expected:} 400 Bad Request \\
\textbf{Actual:} 200 OK

\subsection*{Bug 3: Profile update accepts non-digit phone}
\textbf{Endpoint:} PUT \texttt{/api/v1/profile} \\
\textbf{Request:} Headers: \texttt{X-Roll-Number: 123}, \texttt{X-User-ID: 799}; Body: \{name: Valid Name, phone: 12345abcde\} \\
\textbf{Expected:} 400 Bad Request \\
\textbf{Actual:} 200 OK

\subsection*{Bug 4: Address creation accepts short pincode}
\textbf{Endpoint:} POST \texttt{/api/v1/addresses} \\
\textbf{Request:} Headers: \texttt{X-Roll-Number: 123}, \texttt{X-User-ID: 799}; Body: \{label: HOME, street: 123 Main Street, city: Metropolis, pincode: 12345, is\_default: false\} \\
\textbf{Expected:} 400 Bad Request \\
\textbf{Actual:} 200 OK

\subsection*{Bug 5: Cart add accepts negative quantity}
\textbf{Endpoint:} POST \texttt{/api/v1/cart/add} \\
\textbf{Request:} Headers: \texttt{X-Roll-Number: 123}, \texttt{X-User-ID: 799}; Body: \{product\_id: 1, quantity: -1\} \\
\textbf{Expected:} 400 Bad Request \\
\textbf{Actual:} 200 OK

\subsection*{Bug 6: Loyalty redeem fails with points available}
\textbf{Endpoint:} POST \texttt{/api/v1/loyalty/redeem} \\
\textbf{Request:} Headers: \texttt{X-Roll-Number: 123}, \texttt{X-User-ID: 799}; Body: \{amount: 1\} \\
\textbf{Expected:} 200 OK when points \(\ge 1\) \\
\textbf{Actual:} 400 Bad Request (points reported \(\ge 1\))

\subsection*{Bug 7: Invoice totals incorrect}
\textbf{Endpoint:} GET \texttt{/api/v1/orders/{order\_id}/invoice} \\
\textbf{Request:} Headers: \texttt{X-Roll-Number: 123}, \texttt{X-User-ID: 799} \\
\textbf{Expected:} \(\text{total} = \text{subtotal} + \text{GST}\) \\
\textbf{Actual:} \texttt{total} does not equal \texttt{subtotal + gst} (observed total 0.0 with non-zero subtotal/gst)

\section*{Notes}
Skipped tests are not bugs; they indicate missing data required to execute a test (for example, no inactive products or no orders for the user).
